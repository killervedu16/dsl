def insertion_sort(arr):
    for i in range(1, len(arr)):  # 2 | 3 1 4 5
        j = i                     # 2 3 | 1 4 5

        while arr[j-1] > arr[j] and j > 0:
            # swap
            temp = arr[j-1]
            arr[j-1] = arr[j]
            arr[j] = temp

            j = j - 1

    return arr

def shell_sort(arr):
    n = len(arr)
    gap = n // 2

    while gap > 0:
        for i in range(gap, n):
            temp = arr[i]
            j = i
            while j >= gap and arr[j - gap] > temp:
                arr[j] = arr[j - gap]
                j = j - gap

            arr[j] = temp

        gap = gap // 2

    return arr
